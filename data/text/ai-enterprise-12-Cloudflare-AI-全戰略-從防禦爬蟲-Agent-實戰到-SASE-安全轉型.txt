# Claude Desktop 每次送AI 請求前執行這個Script
# 目的：抓使用者資訊，包成JSON 輸出給Claude Desktop
# 這些資訊最終會出現在Cloudflare AI Gateway Logs
# 你的Cloudflare AI Gateway API Key
$apiKey = "REPLACE_WITH_YOUR_CF_AI_GATEWAY_API_KEY“
# 從Windows 環境變數讀取帳號名稱和電腦名稱
$username = $env:USERNAME
$hostname = $env:COMPUTERNAME
# 執行whoami /all，用regex 從輸出裡找MicrosoftAccount\xxx@xxx.com
# 找不到的話（沒連結Microsoft 帳號）就用帳號名@unknown 代替
$allOutput = (whoami /all 2>&1) -join " "
$emailMatch = [regex]::Match($allOutput, 'MicrosoftAccount\\([^\s]+@[^\s]+)')
if ($emailMatch.Success) { $email = $emailMatch.Groups[1].Value }
else { $email = "$username@unknown" }
# 組成cf-aig-metadata 的內容，引號要跳脫（" → \"）否則JSON 會壞掉
$metadata = "{`"user`": `"$username`", `"email`": `"$email`", `"hostname`": `"$hostname`"}"
$metadataEscaped = $metadata -replace '"', '\“’
# 輸出最終JSON，Claude Desktop 讀到後會用token 驗證、並帶上headers 發請求
Write-Output "{`"token`": `"$apiKey`", `"headers`": {`"cf-aig-metadata`": `"$metadataEscaped`"}}"
